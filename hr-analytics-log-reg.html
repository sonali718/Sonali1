def Age(dataframe):
    dataframe.loc[dataframe["age"]<=30,"age"]=1
    dataframe.loc[(dataframe["age"]>30)&(dataframe["age"]<=40),"age"]=2
    dataframe.loc[(dataframe["age"]>40)&(dataframe["age"]<=50),"age"]=3
    dataframe.loc[(dataframe["age"]>50)&(dataframe["age"]<=60),"age"]=4
